\begin{example}{4}[2025][全国2卷][高三][高考][第19题]

    \item 证明：当$m\in \mathbf{N}^*$时，$P_{2m+1}-q_{2m+1}<P_{2m}-q_{2m}<P_{2m+2}-q_{2m+2}$．
\end{enumerate}
\end{example}
