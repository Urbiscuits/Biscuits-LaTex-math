\renewcommand{\AddImage}[3]{%
	\node[anchor=north east] at #3 {%
		\IfFileExists{image/#1.png}{%
			% PNG 图片存在 → 直接插入
			\includegraphics[width=#2]{image/#1.png}%
		}{%
			% PNG 不存在 → 根据题号绘制 TikZ 图形
			\ifnum #1 = 18
			% ---------- 第18题的 TikZ 示例 ----------
			\begin{tikzpicture}[scale=1.2]
				\draw[->] (-0.5,0) -- (3,0) node[right] {$x$};
				\draw[->] (0,-0.5) -- (0,2.5) node[above] {$y$};
				\draw[domain=0:2, smooth, variable=\x, blue, thick]
				plot ({\x}, {0.5*\x*\x}) node[right] {$y=\frac{1}{2}x^2$};
			\end{tikzpicture}
			\else\ifnum #1 = 19
			% ---------- 第19题（如需） ----------
			\begin{tikzpicture}[scale=1.2]
				\draw[red] (0,0) circle (1);
			\end{tikzpicture}
			\else\ifnum #1 = 20
			% ---------- 第20题（如需） ----------
			\begin{tikzpicture}[scale=1.2]
				\draw[red] (0,0) circle (1);
			\end{tikzpicture}
			\else
			% 其他题号：可留空或再添加条件
			\fi\fi\fi
		}%
	};%
}
